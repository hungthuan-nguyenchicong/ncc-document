## setup-ssh-key
```
setup-ssh-key() {
    SSH_DIR="/root/.ssh"
    KEY_FILE="$SSH_DIR/id_ed25519"

    # 1. Tao thu muc .ssh neu chua co
    if [ ! -d "$SSH_DIR" ]; then
        mkdir -p "$SSH_DIR"
    fi
    # Luon dam bao quyen thu muc la 700
    chmod 700 "$SSH_DIR"

    # 2. Neu chua co Key, tien hanh tao moi
    if [ ! -f "$KEY_FILE" ]; then
        echo "🔑 No SSH Key found. Generating a new ed25519 key..."
        # -N "" de khong hoi passphrase, giup tu dong hoa hoan toan
        ssh-keygen -t ed25519 -C "git@gmail.com" -N "" -f "$KEY_FILE"
        echo "✅ New key generated at $KEY_FILE"
        echo "📜 Your Public Key (dan cai nay vao GitHub):"
        cat "${KEY_FILE}.pub"
    else
        # 3. Neu da co Key (do mount volume hoac da tao truoc do)
        # Bat buoc phai chmod 600 thi SSH moi cho phep su dung key nay
        chmod 600 "$KEY_FILE"
        # Neu co file pub thi cung nen chmod 644
        [ -f "${KEY_FILE}.pub" ] && chmod 644 "${KEY_FILE}.pub"
    fi
}
```
